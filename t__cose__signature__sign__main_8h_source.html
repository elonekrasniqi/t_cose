99/* This is how you get the general interface / instance for a signer,
100 * a t_cose_signer, from the specific and concrete instance of a
101 * signer. Because the t_cose_signer is the first member in a
102 * t_cose_ecdsa_signer, the implementation for this is in essence just
103 * a cast and in the end no code is generated.
104 *
105 * t_cose calls signers as follows:
106 * struct t_cose_signature_sign *signer;
107 * signer = t_cose_signature_sign_from_main(me);
108 *
109 * result = (signer->s.callback)(signer, ....);
110 *
111 * It makes use of the function pointer in signer->s. This callback is
112 * when all the interesting work id done by
113 * t_cose_signature_sign_main.
114 *
115 */
