851static inline bool
852t_cose_params_empty(struct q_useful_buf_c encoded_params)
853{
854 int compare_result;
855 const uint8_t empty_map[] = {0xa0};
856
857 if(t_cose_params_empty_bstr(encoded_params)) {
858 return true;
859 }
860
861 compare_result = q_useful_buf_compare(encoded_params,
862 Q_USEFUL_BUF_FROM_BYTE_ARRAY_LITERAL(empty_map));
863
864 /* Check for a CBOR-encoded empty array */
865 return compare_result == 0 ? true : false;
866}

869static inline struct t_cose_parameter
870t_cose_param_make_alg_id(int32_t alg_id)
871{
872 /* The weird world of initializers, compound literals for
873 * C and C++...
874 *
875 * There are three contexts where it would be nice to assign /
876 * initialize with a filled-in t_cose_parameter.
877 *
878 * 1) Assignments -- an expression is required
879 * 2) Initialization of a variable upon its declaration.
880 * 3) Initialization of a static const data structure -- initialized
881 * data in data sections of executable object code
882 *
883 * The inline functions here provide 1) and 2) for C and C++, but not 3).
884 *
885 * The macros like this
886 * #define T_COSE_MAKE_ALG_ID_PARAM(alg_id) \
887 * {T_COSE_HEADER_PARAM_ALG, \
888 * true,\
889 * false,\
890 * {0,0},\
891 * T_COSE_PARAMETER_TYPE_INT64,\
892 * .value.int64 = alg_id }
893 * is an initializer. Both C and C++ have initializers,
894 * but in C++ they can't use designated initializers,
895 * the part where .value.int64 = alg_id.
896 *
897 * The following is a compound literal.
898 * #define T_COSE_MAKE_ALG_ID_PARAM(alg_id) \
899 * (struct t_cose_parameter){T_COSE_HEADER_PARAM_ALG, \
900 * true,\
901 * false,\
902 * {0,0},\
903 * T_COSE_PARAMETER_TYPE_INT64,\
904 * .value.int64 = alg_id }
905 *
906 * It looks like a cast but it is not. You can take the address of
907 * it and it is an lvalue. These exist only in C, not in C++,
908 * though c++ compilers do support them, but warnings will ensue
909 * if -Wpendatic is used so this code doesn't use them with c++
910 *
911 * https://stackoverflow.com/questions/28116467/are-compound-literals-standard-c
912 *
913 * See also the definition of NULLUsefulBufC in UsefulBuf.h in QCBOR.
914 *
915 * In the end, I expect the optimizer produces good code for all
916 * of these constructs. For example the code produces whether this
917 * is created by a function, an initializer or a compound literal
918 * is hopefully the same.
919 */
920
921 struct t_cose_parameter parameter;
922
923 parameter.critical = false;
924 parameter.in_protected = true;
925 parameter.location.index = 0;
926 parameter.location.nesting = 0;
927 parameter.label = T_COSE_HEADER_PARAM_ALG;
928 parameter.value_type = T_COSE_PARAMETER_TYPE_INT64;
929 parameter.value.int64 = alg_id;
930 parameter.next = NULL;
931
932 return parameter;
933}
