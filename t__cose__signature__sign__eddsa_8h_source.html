141/* This is how you get the general interface / instance for a signer,
142 * a t_cose_signer, from the specific and concrete instance of a
143 * signer. Because the t_cose_signer is the first member in a
144 * t_cose_ecdsa_signer, the implementation for this is in essence just
145 * a cast and in the end no code is generated.
146 *
147 * t_cose calls signers as follows:
148 * struct t_cose_signature_sign *signer;
149 * signer = t_cose_signature_sign_from_eddsa(me);
150 *
151 * result = (signer->s.callback)(signer, ....);
152 *
153 * It makes use of the function pointer in signer->s. This callback is
154 * where all the interesting work is done by
155 * t_cose_signature_sign_eddsa.
156 *
157 */
158static struct t_cose_signature_sign *
159t_cose_signature_sign_from_eddsa(struct t_cose_signature_sign_eddsa *me);
